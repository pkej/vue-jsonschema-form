# Example

const schema = {
  type: 'object',
  properties: {
    a: { type: 'string', title: 'Text Input' },
    b: { type: 'number', title: 'Digits Only' },
    c: { type: 'boolean', default: true }
  }
}

<JsonSchema :schema="schema" v-model="value"/>
